    <script>

        let map;

        let kmzLayer;

        let selectedLayer = null;

        let gsbVisualLayers = [];

        let excelDatabase = [];

        let allRoads = [];

 

        function initMap() {

            map = L.map('map', { zoomControl: false }).setView([-6.2088, 106.8456], 13);

            L.control.zoom({ position: 'bottomright' }).addTo(map);

 

            L.tileLayer('https://{s}.google.com/vt/lyrs=s,h&x={x}&y={y}&z={z}', {

                maxZoom: 20,

                subdomains: ['mt0', 'mt1', 'mt2', 'mt3'],

                attribution: '&copy; Google Satellite'

            }).addTo(map);

        }

 

        const roadSearch = document.getElementById('roadSearch');

        const searchResults = document.getElementById('search-results');

 

        roadSearch.addEventListener('input', (e) => {

            const query = e.target.value.toLowerCase().trim();

            if (query.length < 2) {

                searchResults.classList.add('hidden');

                return;

            }

 

            const matches = allRoads.filter(item =>

                (item.feature.properties.name || "").toLowerCase().includes(query)

            ).slice(0, 10);

 

            if (matches.length > 0) {

                searchResults.innerHTML = matches.map((item, index) => `

                    <div class="p-3 hover:bg-slate-50 cursor-pointer border-b border-slate-100 flex flex-col" onclick="selectFromSearch(${index})">

                        <span class="text-sm font-bold text-slate-800">${item.feature.properties.name || "Jalan Tanpa Nama"}</span>

                        <span class="text-[10px] text-slate-400">Peta: ${item.feature.properties.styleUrl || 'Data KML'}</span>

                    </div>

                `).join('');

                window.currentMatches = matches;

                searchResults.classList.remove('hidden');

            } else {

                searchResults.innerHTML = `<div class="p-3 text-sm text-slate-400">Jalan tidak ditemukan</div>`;

                searchResults.classList.remove('hidden');

            }

        });

 

        function selectFromSearch(index) {

            const match = window.currentMatches[index];

            if (!match) return;

 

            const layer = match.layer;

            const feature = match.feature;

 

            let latlng;

            if (feature.geometry.type === "LineString") {

                const coords = feature.geometry.coordinates;

                const centerIdx = Math.floor(coords.length / 2);

                latlng = L.latLng(coords[centerIdx][1], coords[centerIdx][0]);

            } else {

                latlng = layer.getBounds().getCenter();

            }

 

            map.flyTo(latlng, 18);

            onObjectSelect(layer, feature, latlng);

           

            searchResults.classList.add('hidden');

            roadSearch.value = feature.properties.name || "";

        }

 

        document.addEventListener('click', (e) => {

            if (!roadSearch.contains(e.target) && !searchResults.contains(e.target)) {

                searchResults.classList.add('hidden');

            }

        });

 

        document.getElementById('xlsxInput').addEventListener('change', (e) => {

            const file = e.target.files[0];

            if (!file) return;

 

            const reader = new FileReader();

            reader.onload = (event) => {

                const data = new Uint8Array(event.target.result);

                const workbook = XLSX.read(data, { type: 'array' });

                const worksheet = workbook.Sheets[workbook.SheetNames[0]];

                const rows = XLSX.utils.sheet_to_json(worksheet, { header: 1 });

               

                excelDatabase = rows.slice(1).map(row => ({

                    namaJalan: (row[0] || "").toString().toLowerCase().trim(),

                    gsb: parseFloat(row[1]) || 0

                })).filter(item => item.namaJalan !== "");

 

                updateStatusBar('xlsx-status', `DATA: ${excelDatabase.length} BARIS`, 'bg-emerald-100 text-emerald-700');

            };

            reader.readAsArrayBuffer(file);

        });

 

        document.getElementById('kmzInput').addEventListener('change', async (e) => {

            const file = e.target.files[0];

            if (!file) return;

 

            const reader = new FileReader();

            if (file.name.endsWith('.kmz')) {

                reader.onload = async (event) => {

                    const zip = await JSZip.loadAsync(event.target.result);

                    const kmlFile = Object.keys(zip.files).find(n => n.endsWith('.kml'));

                    const kmlText = await zip.files[kmlFile].async("string");

                    loadGeoJson(kmlText);

                };

                reader.readAsArrayBuffer(file);

            } else {

                reader.onload = (event) => loadGeoJson(event.target.result);

                reader.readAsText(file);

            }

        });

 

        function loadGeoJson(xmlText) {

            const parser = new DOMParser();

            const xml = parser.parseFromString(xmlText, "text/xml");

            const geojson = toGeoJSON.kml(xml);

 

            if (kmzLayer) map.removeLayer(kmzLayer);

            allRoads = [];

 

            kmzLayer = L.geoJSON(geojson, {

                style: { color: "#fbbf24", weight: 4, opacity: 0.6 },

                onEachFeature: (feature, layer) => {

                    allRoads.push({ feature, layer });

                    layer.on('click', (e) => {

                        onObjectSelect(layer, feature, e.latlng);

                        L.DomEvent.stopPropagation(e);

                    });

                }

            }).addTo(map);

 

            const bounds = kmzLayer.getBounds();

            if (bounds.isValid()) map.fitBounds(bounds);

            updateStatusBar('kmz-status', "MAP: DIMUAT", 'bg-amber-100 text-amber-700');

        }

 

        function clearGsbVisuals() {

            gsbVisualLayers.forEach(l => map.removeLayer(l));

            gsbVisualLayers = [];

        }

 

        function onObjectSelect(layer, feature, latlng) {

            if (selectedLayer) kmzLayer.resetStyle(selectedLayer);

            clearGsbVisuals();

           

            selectedLayer = layer;

            layer.setStyle({ color: '#ffffff', weight: 6, opacity: 0.9 });

            layer.bringToFront();

 

            const props = feature.properties;

            const roadName = (props.name || "Tanpa Nama").trim();

            const roadNameLower = roadName.toLowerCase();

 

            const matchedExcel = excelDatabase.find(item =>

                roadNameLower.includes(item.namaJalan) || item.namaJalan.includes(roadNameLower)

            );

 

            let finalGsb = 0;

            let isExcelSource = false;

 

            if (matchedExcel) {

                finalGsb = matchedExcel.gsb;

                isExcelSource = true;

            } else {

                finalGsb = parseFloat(props.gsb || props.GSB || props.sempadan) || 0;

            }

 

            if (finalGsb > 0 && feature.geometry.type === "LineString") {

                drawGsbLines(feature, finalGsb);

            }

 

            updateUI(roadName, finalGsb, isExcelSource, latlng, props);

        }

 

        function drawGsbLines(feature, gsbValue) {

            const line = feature.geometry;

            const distanceKm = gsbValue / 1000;

 

            try {

                const leftOffset = turf.lineOffset(line, distanceKm, { units: 'kilometers' });

                const leftLayer = L.geoJSON(leftOffset, {

                    style: { color: '#10b981', weight: 2, dashArray: '5, 10', opacity: 0.8 }

                }).addTo(map);

 

                const rightOffset = turf.lineOffset(line, -distanceKm, { units: 'kilometers' });

                const rightLayer = L.geoJSON(rightOffset, {

                    style: { color: '#10b981', weight: 2, dashArray: '5, 10', opacity: 0.8 }

                }).addTo(map);

 

                const buffered = turf.buffer(line, distanceKm, { units: 'kilometers' });

                const bufferLayer = L.geoJSON(buffered, {

                    style: { color: '#10b981', weight: 0, fillOpacity: 0.1 }

                }).addTo(map);

 

                gsbVisualLayers.push(leftLayer, rightLayer, bufferLayer);

            } catch (err) {

                console.error("Gagal menggambar proyeksi GSB:", err);

            }

        }

 

        function updateUI(name, gsb, isExcel, latlng, props) {

            document.getElementById('welcome-state').classList.add('hidden');

            document.getElementById('info-state').classList.remove('hidden');

           

            document.getElementById('road-name-display').innerText = name;

            document.getElementById('gsb-val').innerText = gsb || "--";

            document.getElementById('coord-txt').innerText = `${latlng.lat.toFixed(5)}, ${latlng.lng.toFixed(5)}`;

           

            const badge = document.getElementById('data-source-badge');

            const alert = document.getElementById('sync-alert');

            const desc = document.getElementById('gsb-desc');

                badge.innerText = "SOURCE: EXCEL";
